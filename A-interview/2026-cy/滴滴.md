# 滴滴

7 种基本类型（String/Number/Boolean/Undefined/Null/Symbol/BigInt）



const arr = [1, 2, 5, 4] 
function aSort(arr) {
    if (arr.length <= 1) {
        return arr;
    }
    const left = [];
    const right = [];
    const len = arr.length;
    const temp = arr[Math.floor(len / 2)-1]; // 取整
    arr.forEach(item => {
        if (item < temp) {
            left.push(item);
        } else if (item > temp) {
            right.push(item);
        }
    })

    return [...aSort(left), temp, ...aSort(right)];
}

aSort(arr); // [1, 2, 4, 5]
