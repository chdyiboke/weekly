function main() {
  console.log(1);
  setTimeout(() => {
    console.log(2);
  }, 1000);
  setTimeout(() => {
    console.log(3);
    Promise.resolve(4).then((res) => {
      console.log(res);
    })
  }, 500);
  try {
    new Promise((resolve, reject) => {
      console.log(5);
      reject(6);
      console.log(7);
      resolve(8);
    }).then((res) => {
      console.log(res)
    }, (err) => {
      console.log('6666', err)
    })
  } catch (error) {
    console.log("error", error)
  }
  console.log(9);
}
