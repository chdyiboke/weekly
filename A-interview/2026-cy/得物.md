function getNumber(n) {
  if (n === 0) {
    return 1;
  }

  return Math.pow(9, n) + getNumber(n - 1);
}

n=0 → 1 （只有数字 0）

n=2 → 1+9+81=91 （0~99 之间无重复数字的数）
n=3 → 1+9+81+729=820 （0~999 之间无重复数字的数）

function getNumber(n) {
  let res = 1;
  let sum = 0;
  for(let i=1; i<=n; i++){
    sum += Math.pow(9,i);
  }
  return res + sum;
}
迭代版：时间 O(n)，空间 O(1) ↑
