class RequestScheduler {
  /**
   * 构造请求调度器
   * @param {number} maxConcurrency - 最大并发数
   */
  constructor(maxConcurrency) {
    // 最大并发数
    this.maxConcurrency = Math.max(1, Number(maxConcurrency)) || 2;
    // 当前正在执行的请求数
    this.currentCount = 0;
    // 等待队列：存储待执行的任务
    this.queue = [];
  }

  /**
   * 添加异步任务到调度器
   * @param {Function} task - 异步任务函数（必须返回 Promise）
   * @returns {Promise} 任务执行结果
   */
  add(task) {
    return new Promise((resolve, reject) => {
      // 将任务和回调存入队列
      this.queue.push({ task, resolve, reject });
      // 尝试执行任务
      this.run();
    });
  }

  /**
   * 执行任务（核心调度逻辑）
   * 自动判断：是否达到最大并发 → 有空闲则执行队列任务
   */
  run() {
    // 无等待任务 / 已达最大并发 → 停止调度
    if (this.queue.length === 0 || this.currentCount >= this.maxConcurrency) {
      return;
    }

    // 取出队首任务
    const { task, resolve, reject } = this.queue.shift();
    // 并发数+1
    this.currentCount++;

    // 执行异步任务
    task()
      .then(resolve)
      .catch(reject)
      .finally(() => {
        // 任务完成（成功/失败）：并发数-1，继续调度下一个
        this.currentCount--;
        this.run();
      });
  }

  // ============= 增强工具方法 =============
  /** 获取等待队列长度 */
  getQueueLength() {
    return this.queue.length;
  }

  /** 获取当前活跃并发数 */
  getActiveCount() {
    return this.currentCount;
  }

  /** 清空等待队列 */
  clearQueue() {
    this.queue = [];
  }
}
