// 深度优先遍历
function maxWidth (tree){
    if (!tree.value) return 0;
    const len = tree.left && tree.right ? 2 : tree.left || tree.right ? 1 : 0; 
    //哈希表记录每层
    return maxWidth(tree.left) + maxWidth(tree.right) + len;
}
